Table --- Addresses

  • AddressID -- Int32 (key)
        Contient l'identifiant unique sur chaque ligne de la table Adresses
  • DBCommand -- DBCommandEnum (int)
        Identifie la command de base de données pour syncroniser avec la base de données local
  • AddressTVItemID -- Int32
        Lien à la table TVItems avec l'identifiant unique
  • AddressType -- AddressTypeEnum (int)
        Type d'adresse
  • CountryTVItemID -- Int32
        Lien à la table TVItems représentant le pays
  • ProvinceTVItemID -- Int32
        Lien à la table TVItems représentant la province
  • MunicipalityTVItemID -- Int32
        Lien à la table TVItems représentant la municipalité
  • StreetName -- String (max=200)
        Nom de la rue de l'adresse
  • StreetNumber -- String (max=50)
        Numéro de la rue de l'adresse
  • StreetType -- StreetTypeEnum (int)
        Type de rue de l'adresse
  • PostalCode -- String (min=6) (max=11)
        Code postale de l'adresse
  • GoogleAddressText -- String (min=10) (max=200)
        Adresse Google fourni par le service Google Location
  • LastUpdateDate_UTC -- DateTime
        Contient la date de la dernière mise à jour en (UTC)
  • LastUpdateContactTVItemID -- Int32
        Contact TVItemID qui a fait le dernier changement
