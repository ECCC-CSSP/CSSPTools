Table --- AppTasks

  • AppTaskID -- Int32 (key)
        Contains the unique identifier on each row of the AppTasks table
  • DBCommand -- DBCommandEnum (int)
        Identify the database command for local database sync
  • TVItemID -- Int32
        Link to the TVItems table with the unique identifier
  • TVItemID2 -- Int32
        Link #2 to the TVItems table with the unique identifier
  • AppTaskCommand -- AppTaskCommandEnum (int)
        App task command
  • AppTaskStatus -- AppTaskStatusEnum (int)
        App task status
  • PercentCompleted -- Int32 (min=0) (max=100)
        App task percent completed
  • Parameters -- String
        App task parameters
  • Language -- LanguageEnum (int)
        App task Language
  • StartDateTime_UTC -- DateTime
        App task start date and time
  • EndDateTime_UTC -- DateTime
        App task end date and time
  • EstimatedLength_second -- Int32 (min=0) (max=1000000)
        App task estimated time (secondes)
  • RemainingTime_second -- Int32 (min=0) (max=1000000)
        App task time remaining (secondes)
  • LastUpdateDate_UTC -- DateTime
        Contains the date of the last update in UTC
  • LastUpdateContactTVItemID -- Int32
        Contact TVItemID who made the last changes
