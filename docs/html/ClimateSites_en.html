Table --- ClimateSites

  • ClimateSiteID -- Int32 (key)
        Contains the unique identifier on each row of the ClimateSites table
  • DBCommand -- DBCommandEnum (int)
        Identify the database command for local database sync
  • ClimateSiteTVItemID -- Int32
        Link to the TVItems table with the unique identifier
  • ECDBID -- Int32 (min=1) (max=100000)
        ECDBID --- Environment Canada weather office web site unique id for the weather climate site
  • ClimateSiteName -- String (max=100)
        Climate site name is a unique name for the site given by Environment Canada
  • Province -- String (max=4)
        Province identified by two letters initiales
  • Elevation_m -- Double (min=0) (max=10000)
        Elevation (m) is the elevation above mean sea level
  • ClimateID -- String (max=10)
        Climate site ID is a unique ID for the climate site for Environment Canada
  • WMOID -- Int32 (min=1) (max=100000)
        WMOID is a unique ID for the climate site for World Meteorological Office
  • TCID -- String (max=3)
        TCID is the identifier assigned by Transport Canada to identify meteorological reports from airport observing sites transmitted in real time in aviation formats.
  • IsQuebecSite -- Boolean
        Quebec Climate Site
  • IsCoCoRaHS -- Boolean
        CoCoRaHS Climate Site
  • TimeOffset_hour -- Double (min=-10) (max=0)
        Time zone offset (hour)
  • File_desc -- String (max=50)
        File description --- temporary field used while uploading data from Environment Canda weather office web site
  • HourlyStartDate_Local -- DateTime
        Hourly start date will contain null is the climate site does not have hourly data or the starting date of hourly data
  • HourlyEndDate_Local -- DateTime
        Hourly end date will contain null is the climate site does not have hourly data or the end date of hourly data
  • HourlyNow -- Boolean
        Hourly now indicates if there are hourly data at this date
  • DailyStartDate_Local -- DateTime
        Daily start date will contain null is the climate site does not have daily data or the start date of hourly data
  • DailyEndDate_Local -- DateTime
        Daily end date will contain null is the climate site does not have daily data or the end date of hourly data
  • DailyNow -- Boolean
        Daily now indicates if there are daily data at this date
  • MonthlyStartDate_Local -- DateTime
        Monthly start date will contain null is the climate site does not have monthly data or the start date of monthly data
  • MonthlyEndDate_Local -- DateTime
        Monthly end date will contain null is the climate site does not have monthly data or the end date of monthly data
  • MonthlyNow -- Boolean
        Monthly now indicates if there are monthly data at this date
  • LastUpdateDate_UTC -- DateTime
        Contains the date of the last update in UTC
  • LastUpdateContactTVItemID -- Int32
        Contact TVItemID who made the last changes
