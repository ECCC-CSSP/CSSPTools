Table --- ClimateSites

  • ClimateSiteID -- Int32 (key)
        Contient l'identifiant unique sur chaque ligne de la table ClimateSites
  • DBCommand -- DBCommandEnum (int)
        Identifie la command de base de données pour syncroniser avec la base de données local
  • ClimateSiteTVItemID -- Int32
        Lien à la table TVItems avec l'identifiant unique
  • ECDBID -- Int32 (min=1) (max=100000)
        ECDBID --- Site Web du bureau météorologique d'Environnement Canada identifiant le site de site climatatique
  • ClimateSiteName -- String (max=100)
        Nom du site climatique est un nom unique pour le site donné par Environnement Canada
  • Province -- String (max=4)
        Province identifiée par les initiales de deux lettres
  • Elevation_m -- Double (min=0) (max=10000)
        Élévation (m) est l'élévation au dessus du niveau moyen de la mer
  • ClimateID -- String (max=10)
        Identifiant du site climatique est unique pour le site climatique pour Environnemement Canada
  • WMOID -- Int32 (min=1) (max=100000)
        WMOID est un identifiant unique pour le site climatique pour World Meteorological Office
  • TCID -- String (max=3)
        TCID est l'identificateur attribué par Transports Canada pour identifier les rapports météorologiques provenant des stations localisées aux aéroports et sont transmis en temps réel dans des formats d'aviation.
  • IsQuebecSite -- Boolean
        Site climatique du Québec
  • IsCoCoRaHS -- Boolean
        Site climatique de CoCoRaHS
  • TimeOffset_hour -- Double (min=-10) (max=0)
        Décalage du fuseau horaire (heure)
  • File_desc -- String (max=50)
        Description de filière --- champ temporaire utilisé lors du téléchargement des données du site Web du bureau météorologique d'Environnement Canada
  • HourlyStartDate_Local -- DateTime
        Date de départ des données horaires contiendra null si le site climatique n'a pas de données horaires ou la date de départ des données horaires
  • HourlyEndDate_Local -- DateTime
        Date de fin des données horaires contiendra null si le site climatique n'a pas de données horaires ou la date de fin des données horaires
  • HourlyNow -- Boolean
        Horaire maintenant indique s'il y a des données horaires maintenant
  • DailyStartDate_Local -- DateTime
        Date de départ des données journalières contiendra null si le site climatique n'a pas de données journalières ou la date de départ des données journalières
  • DailyEndDate_Local -- DateTime
        Date de fin des données journalières contiendra null si le site climatique n'a pas de données journalières ou la date de fin des données journalières
  • DailyNow -- Boolean
        Journalier maintenant indique s'il y a des données journalières maintenant
  • MonthlyStartDate_Local -- DateTime
        Date de départ des données mensuelles contiendra null si le site climatique n'a pas de données mensuelles ou la date de départ des données mensuelles
  • MonthlyEndDate_Local -- DateTime
        Date de fin des données mensuelles contiendra null si le site climatique n'a pas de données mensuelles ou la date de fin des données mensuelles
  • MonthlyNow -- Boolean
        Mensuel maintenant indique s'il y a des données mensuelles maintenant
  • LastUpdateDate_UTC -- DateTime
        Contient la date de la dernière mise à jour en (UTC)
  • LastUpdateContactTVItemID -- Int32
        Contact TVItemID qui a fait le dernier changement
