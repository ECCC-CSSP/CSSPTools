Table --- MikeSources

  • MikeSourceID -- Int32 (key)
        Contains the unique identifier on each row of the MikeSources table
  • DBCommand -- DBCommandEnum (int)
        Identify the database command for local database sync
  • MikeSourceTVItemID -- Int32
        Link to the TVItems table representing MIKE source
  • IsContinuous -- Boolean
        Source is continuous
  • Include -- Boolean
        Source is included
  • IsRiver -- Boolean
        Source is a river
  • UseHydrometric -- Boolean
        Source should use hydrometric site with factor for discharge
  • HydrometricTVItemID -- Int32
        Link to the TVItems table representing hydrometric site
  • DrainageArea_km2 -- Double (min=0) (max=1000000)
        Drainage area in square kilometers
  • Factor -- Double (min=0) (max=1000000)
        Factor
  • SourceNumberString -- String (max=50)
        Source number
  • LastUpdateDate_UTC -- DateTime
        Contains the date of the last update in UTC
  • LastUpdateContactTVItemID -- Int32
        Contact TVItemID who made the last changes
